DESeq2

DESeq2 – это царь-библиотека необходимая для работы с данными RNA-seq. Тут и дифференциальная экспрессия, и нормализации, и PCA-plots.

строим PCA

dds <- DESeqDataSetFromMatrix(countData = counts,
                              colData = conditions,
                              design = ~ cells + conditions)
ind <- order(rowSums(counts(dds)), decreasing = TRUE)
dds <- dds[ind, ][1:8000,]
dds <- DESeq(dds)
vst_dds <- vst(dds)
counts.norm <- assay(vst_dds)



plotPCA(vst_dds, intgroup=c("cells", "conditions")) + theme_bw()

Kmeans

Используем rlog(dds) для получения логарифмированных значений экспрессии

rlog_dds <- rlog(dds)
counts.log2  <- assay(rlog_dds)

Используем функцию Kmeans из пакета amap с методом “correlation”, c количеством кластеров равным 6. (Вы можете выбрать своё количество кластеров), и максимальным количеством итераций равным 20000

clustering <- Kmeans(counts.log2, 8, method="correlation", iter.max=20000)

После этого вы можете достать соотвествие каждого гена кластеру с помощью элемента листа с названием “cluster”

head(clustering$cluster)

После этого от вас потребуется эту кластеризацию изобразить на хитмапе: гены должны быть отсортированы по кластерам, а столбцы сначала по клеткам, а затем по тканям.

to_visualise <- counts.log2[order(clustering$cluster), order(conditions$cells, conditions$conditions)]
cluster <- as.factor(clustering$cluster)
cluster <- as.data.frame(cluster)
to_visualise <- t(apply(to_visualise, 1, function(r) {
  (r - min(r)) / (max(r) - min(r))
}))


pheatmap(to_visualise, 
         show_rownames = F, cluster_rows = F,
         cluster_cols=F,
         annotation_col = conditions,
         annotation_row =  cluster)

Самый обычный способ визуализировать данные дифф.экспрессии – это volcano plot. По оси x мы будем откладывать log fold change, а по y - adjusted p value.

# Строим Volcano plot

gdata1 <- data.frame(
  x=res_FB$log2FoldChange,
  y=-log10(res_FB$padj),
  name_ =  "1"
)

gdata2 <- data.frame(
  x=res_FAP$log2FoldChange,
  y=-log10(res_FAP$padj),
  name_ =  "2"
)


gdata3 <- data.frame(
  x=res_SMP$log2FoldChange,
  y=-log10(res_SMP$padj),
  name_ =  "3"
)


gdata4 <- rbind(gdata1, gdata2, gdata3)
res4 <- rbind(res_FB, res_FAP, res_SMP)

gdata4 <- gdata4 %>%
  mutate(color = ifelse(res4$padj > 0.01, 
                        yes = "Significant", 
                        no =  "Not Significant"))



ggplot(data=gdata4, aes(x=x, y=y)) +
  geom_point(aes(color = factor(color)), size=1) + theme_bw()  +
  xlab("Log fold change") +
  ylab("Adjusted p.value") +
  geom_hline(yintercept=2, color='red') +
  facet_grid(. ~ name_) +
  scale_color_manual(values = c("Significant" = "#000000", 
                                "Not Significant" = "#ff0000")) # change colors
